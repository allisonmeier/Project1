A - inner =  8.5 AU, outer = 12.5 AU
F - inner = 1.5 AU, outer = 2.2 AU
G - inner = 0.95 AU, outer = 1.4 AU
K - inner = 0.38 AU, outer = 0.56 AU
M - inner = 0.08 AU, outer = 0.12 AU
